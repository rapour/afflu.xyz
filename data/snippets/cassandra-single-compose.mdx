---
title: 'Cassandra setup'
description: 'A bare minimun docker compose file for single node Cassandra and its client'
logo: '/static/images/snippets/Go-Logo.png'
---

```yml
version: "3.9"
services:
  cassandra:
      image: cassandra:latest
      container_name: cassandra
      ports:
        - "9042:9042"
      environment:
        - "MAX_HEAP_SIZE=256M"
        - "HEAP_NEWSIZE=128M"
      restart: always
      volumes:
        - ./out/cassandra_data:/var/lib/cassandra
 
  # access client with the following command: 
  # docker exec -it client bash
  client: 
      image: cassandra:latest
      container_name: client
```
